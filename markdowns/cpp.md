## Checking the sample code

```C++ runnable
#include <iostream>
#include <bitset>
using namespace std;
int main()
{
    string m;
    // getline(cin, m);
    m = "CG";
    string c[2] = {"00", "0"};
    string b;
    for (auto & ch: m)
        b += bitset<7>(ch).to_string();
    string a;
    a += c[b[0] - '0'] + " 0";
    for (int i = 1; i < b.length(); i++) {
        if (b[i] == b[i - 1])
            a += "0";
        else
            a += " " + c[b[i] - '0'] + " 0";
    }
    // cerr << "Debug messages..." << endl;
    cout << a << endl;
}
```
